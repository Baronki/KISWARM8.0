### 7.2 Drift Calculation

```python
def calculate_drift(self, old_state: dict, new_state: dict) -> float:
    """
    Measure real state mutation using cosine similarity.
    0.0 = no change, 1.0 = complete transformation
    """
    # Serialize states to JSON strings
    old_vec = np.array([ord(c) for c in json.dumps(old_state, sort_keys=True)])
    new_vec = np.array([ord(c) for c in json.dumps(new_state, sort_keys=True)])
    
    # Pad to same length
    max_len = max(len(old_vec), len(new_vec))
    old_vec = np.pad(old_vec, (0, max_len - len(old_vec)))
    new_vec = np.pad(new_vec, (0, max_len - len(new_vec)))
    
    # Cosine similarity
    similarity = np.dot(old_vec, new_vec) / (np.linalg.norm(old_vec) * np.linalg.norm(new_vec))
    
    return float(1.0 - similarity)  # drift = 1 - similarity
```
